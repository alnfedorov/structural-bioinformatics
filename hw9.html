smiles = [
    ('OH', 'OC(=O)NC1C(C(C(OC1O)CO)O)O'),
    ('NH3+', '[N+]C(=O)NC1C(C(C(OC1O)CO)O)O'),
    ('H', 'C(=O)NC1C(C(C(OC1O)CO)O)O'),
    ('PH', 'c1ccccc1C(=O)NC1C(C(C(OC1O)CO)O)O'),
    ('COO', '[O-]C(=O)C(=O)NC1C(C(C(OC1O)CO)O)O')
]

mols = []
images = []

for r, s in smiles:
    m = oddt.toolkit.readstring('smi', s)
    if not m.OBMol.Has3D(): 
        m.make3D(forcefield='mmff94', steps=150)
        m.removeh()
        m.OBMol.AddPolarHydrogens()
    m.name = r
    mols.append(m)
    images.append((SVG(copy.deepcopy(m).write('svg'))))

display_svg(*images)
