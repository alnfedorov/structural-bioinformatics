def orca_workload(length=1.52986, hcc=111.200, tors=180):
    return TEMPLATE.format(length=length, hcc=hcc, tors1=tors, tors2=tors-60, tors3=tors+60)

def run_orca(workload):
    with open("/tmp/orca.inp", 'w') as file:
        file.write(workload)
    
    p = subprocess.Popen("/home/shad/progs/bin/orca /tmp/orca.inp", 
                         shell=True, stdout=subprocess.PIPE, stderr=subprocess.PIPE)
    out = p.communicate()[0].decode()
    out = [x for x in out.splitlines() if "FINAL SINGLE POINT ENERGY" in x]
    assert len(out) == 1
    energy = out[0].lstrip("FINAL SINGLE POINT ENERGY").strip()
    return float(energy)

def fit(x, y, tofit, init_params):
    loss = lambda params, _x, _y: tofit(params, _x) - _y
    
    params, _ = optimize.leastsq(loss, init_params[:], args=(x, y))
    return params, lambda x: tofit(params, x)


def plot_fitted(x, y, func):
    plt.scatter(x, y, color='blue', alpha=0.5)
    
    linx = np.linspace(min(x), max(x), 100)
    predy = func(linx)
    plt.plot(linx, predy, color='green')
    
    plt.grid()
    
    plt.margins(y = 0.005 * (max(y) - min(y)))
    
    print("MSE: {0}".format(np.square(y - func(x)).mean()))
